Step 1: Update your Raspberry Pi
   sudo apt update && sudo apt upgrade -y

Step 2: Install Git and Docker
   curl -fsSL https://get.docker.com -o get-docker.sh
   sudo sh get-docker.sh
   sudo usermod -aG docker $USER
   sudo apt install -y git

Step 3: Reboot your Pi (Important for Docker permissions)
   sudo reboot

Step 4: Clone the repository
   git clone https://github.com/EdiynB/Tranga-App.git
   cd Tranga-App

Step 5: Get your Pi's IP address
   hostname -I

Step 6: Start the application
   # Replace <YOUR_PI_IP> with the IP address from Step 5
   # Example: RPi_IP=192.168.1.50 docker compose up -d --build
   RPi_IP=<YOUR_PI_IP> docker compose up -d --build

Step 7: Access the app
   Open a browser on your computer and go to: http://<YOUR_PI_IP>
